let times = 0;

  str += '";';

  const script = new (require('vm').Script)(str);

}

(async () => {
  while (true) {
    run();

    await new Promise(resolve => {
      setTimeout(() => {
        resolve();
      }, 10);
    });

    gc();
  }
})();
